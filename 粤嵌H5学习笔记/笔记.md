function getNextElement(node) { /* 查找下一个元素节点 */
    if (node.nodeType == 1) {
        return node;
    }
    if (node.nextSibling) {
        return getNextElement(node.nextSibling);
    }
    return null;
}

function getNextElementOf(node, name) { /* 查找下一个指定类型元素节点 */
    if (node.nodeType == 1 && name ? node.nodeName == name.toLowerCase() : true) {
        return node;
    }
    if (node.nextSibling) {
        return getNextElementOf(node.nextSibling, name);
    }
    return null;
}
